export const AnnotationPointCondition = memo(() => {

      },

        selector: [

            operator: 'in',
            value: '2023',
          },
        ],
        text: '2023 年',
        textAlign: 'left',
        offsetX: 0,
      },
    ],
    dimensions: [{ id: 'date', alias: '日期',   }],
    measures: [
      { id: 'profit', alias: '利润' },
      { id: 'sales', alias: '销售额' },
      { id: 'count', alias: '数量' },
    ],
  }
  return <VChartRender vseed={vseed} />
})
